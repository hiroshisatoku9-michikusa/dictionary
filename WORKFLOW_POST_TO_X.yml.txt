name: Post dictionary entry to X

on:
  schedule:
    # JST 00:00, 08:00, 21:00
    # Chosen to cover US morning, US evening, and Europe lunch/afternoon windows.
    - cron: "0 12,15,23 * * *"
  workflow_dispatch:

permissions:
  contents: write

concurrency:
  group: post-dictionary-entry-to-x
  cancel-in-progress: false

jobs:
  post:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v4
        with:
          fetch-depth: 0

      - name: Post next dictionary entry
        env:
          X_API_KEY: ${{ secrets.X_API_KEY }}
          X_API_SECRET: ${{ secrets.X_API_SECRET }}
          X_ACCESS_TOKEN: ${{ secrets.X_ACCESS_TOKEN }}
          X_ACCESS_TOKEN_SECRET: ${{ secrets.X_ACCESS_TOKEN_SECRET }}
        run: python3 scripts/post_to_x.py

      - name: Commit post state
        run: |
          if git diff --quiet post_state.json; then
            echo "No state change."
            exit 0
          fi
          git config user.name "github-actions[bot]"
          git config user.email "41898282+github-actions[bot]@users.noreply.github.com"
          git add post_state.json
          git commit -m "Advance X post state"
          git push
